python自动化办公

最近有个伙计考事业单位，有个题库，密密麻麻的，让我把题目和答案分类别保存出来，我一看这么题目，肯定不能手动给他干，就手动写个脚本处理一把

技术：python, docx中文格式处理，excel读操作，读sheet

不足：略微有点儿冗余代码可以优化

from openpyxl import load_workbook
from docx.shared import Pt
from docx.oxml.ns import qn

# 单选题 多选题
def dan_xuan(wb, sheet_name):
    sheet = wb.get_sheet_by_name(sheet_name)
    from docx import Document
    doc_question = Document()
    doc_answer = Document()
    # 设置样式
    doc_question.styles['Normal'].font.name = 'Times New Roman'
    doc_question.styles['Normal']._element.rPr.rFonts.set(qn('w:eastAsia'), u'宋体')

    doc_answer.styles['Normal'].font.name = 'Times New Roman'
    doc_answer.styles['Normal']._element.rPr.rFonts.set(qn('w:eastAsia'), u'宋体')

    # 追加
    p_question = doc_question.add_paragraph('')
    p_answer = doc_answer.add_paragraph('')
    answers = []
    for lie, row in enumerate(sheet.rows):
        for index, cell in enumerate(row):
            if index == 1:
                question = str(lie) + ". " + str(cell.value + "\n")
                run_question = p_question.add_run(question)
                run_question.bold = True
                run_question.font.size = Pt(10)
            if index == 2:
                run_question = p_question.add_run("A " + str(cell.value) + "\r")
                run_question.font.size = Pt(10)
            if index == 3:
                run_question = p_question.add_run("B " + str(cell.value) + "\n")
                run_question.font.size = Pt(10)
            if index == 4:
                run_question = p_question.add_run("C " + str(cell.value) + "\r")
                run_question.font.size = Pt(10)
            if index == 5:
                run_question = p_question.add_run("D " + str(cell.value) + "\n")
                run_question.font.size = Pt(10)
            if index == 6:
                # 答案
                answers.append(str(cell.value))

    # 处理答案
    zhui = ""
    for index, answer in enumerate(answers):
        jia = str(index) + "." + answer + " " + " " + " " + " "
        zhui += jia
        if index % 5 == 0:
            p_answer.add_run(zhui + "\n")
            zhui = ""

    doc_question.save("./li/" + sheet_name + "_问题.docx")
    doc_answer.save("./li/" + sheet_name + "_答案.docx")
