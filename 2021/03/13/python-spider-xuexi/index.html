抓取学习**的题目

2021-03-13
python爬虫

项目开发是有盆友想抓取学习strong country的题目，项目采用selenium手动模拟，再用etree分析，抓取下来

项目难点是更新 同一个selenium 访问不同的 url ，源码直接如下，


from selenium import webdriver
from time import sleep
from lxml import etree
from selenium.webdriver.support.wait import WebDriverWait
from selenium.webdriver.support import expected_conditions as EC
from selenium.webdriver.common.by import By

'''
https://blog.csdn.net/sinat_42349312/article/details/83653342
'''

Chrome_Options = webdriver.ChromeOptions()
Chrome_Options.add_argument("user-data-dir=selenium")
driver = webdriver.Chrome(chrome_options=Chrome_Options)

'''
思路：
    判断是填空题还是选择题
    1. 获得题目内容
    2. 判断题型
        2.1 如果是填空题 --> 给填空一个答案
        2.2 如果是选择题 --> 默认选A  ——>获得四个选项
    3. 保存
        3.1 如果是填空题 --> 保存在 填空题文档
        3.2 如果是选择题 --> 保存在 选择题文档
            3.2.1 如果是多选题  --> 默认选A
            3.2.2 如果是单选题  --> 默认选A
    4. 点击 下一个题目  sleep 如果没有点击下一页手动下一页


更新 同一个selenium 访问不同的 url 
'''


def writeType(filename, questions, answers):
    with open(filename, 'a', encoding='utf-8') as f:
        for question in questions:
            f.write(question + '\n')
        for answer in answers:
            f.write(answer + '\n')
        f.write("\n\n")


def write_URL(url):
    with open("填空.txt", 'a', encoding='utf-8') as f:
        f.write(url + '\n')
    f.close()
    with open("多选题.txt", 'a', encoding='utf-8') as f:
        f.write(url + '\n')
    f.close()
    with open("单选题.txt", 'a', encoding='utf-8') as f:
        f.write(url + '\n')
    f.close()



url = 'https://pc.xuexi.cn/points/exam-paper-detail.html?id=299'
driver.get(url)
sleep(7)

# [345, 343, 342, 341, 340, 339, 338, 337, 300, 301, 302]

html = driver.page_source
page = 0
locator = (By.ID, "app")
# 不想写入文件是哪个链接下的题目 把这一行注释了
print("url: ", url)
write_URL(url)
while page < 10:  # 共10页，这里是手工指定的
    WebDriverWait(driver, 10, 0.5).until(
        EC.presence_of_element_located(locator))
    # 获取源代码
    html = driver.page_source
    e = etree.HTML(html)
    # 题型
    questionTypes = e.xpath('//div[@class="question"]/div[@class="q-header"]/text()')[0]
    questionTypes = str(questionTypes)
    # print(type(questionTypes))
    # print("题型：", questionTypes)
    print("page :", page, "题型：", questionTypes)
    if "填空" in questionTypes:
        # 填空题
        gapFilling = e.xpath('//div[@class="question"]/div[@class="q-body"]/div/span/text()')
        # print("填空题题目：", gapFilling)
        with open('填空.txt', 'a', encoding='utf-8') as f:
            for data in gapFilling:
                f.write(data + '\n')
            f.write("\n\n")
        driver.find_element_by_xpath('//input[@class="blank"]').send_keys("python")  # 填空写个东西
        # elif "选" in questionTypes
    else:
        # 多选题选择
        questions = e.xpath('//div[@class="question"]/div[@class="q-body"]/div/text()')
        # print("选择题题目：", questions)
        answerS = e.xpath('//div[@class="question"]/div[@class="q-answers"]/div/text()')  # 获取选择题的选项
        if "多选题" in questionTypes:
            writeType("多选题.txt", questions, answerS)
        else:
            writeType("单选题.txt", questions, answerS)
        # 如果是选择题默认选A
        driver.find_element_by_xpath(
            '//div[@class="question"]/div[@class="q-answers"]/div[1]').click()

    # 页面跳转
    sleep(2)
    driver.find_element_by_xpath('//button[@class="ant-btn next-btn ant-btn-primary"]').click()
    page = page + 1
    sleep(2)

# 关闭页面
driver.execute_script('window.close()')
print("自动关闭")
driver.quit()
